let lastX = 0, lastY = 0; let wordIndex = -1; const genesisWords = [ "In the beginning", "God", "created", "heavens", "earth", "formless", "void", "darkness", "deep", "Spirit", "hovering", "waters" ]; let particles = []; let hoverTime = 0; // how long mouse/finger has stayed still let lastMouseX = 0; let lastMouseY = 0; class Particle { constructor() { this.x = random(width); this.y = random(height); this.vx = random(-0.8, 0.8); this.vy = random(-0.8, 0.8); this.size = random(2, 5); this.hue = random(190, 240); // deep ocean / void colors } update() { this.x += this.vx; this.y += this.vy; // gentle wrap-around (feels endless) if (this.x < 0) this.x = width; if (this.x > width) this.x = 0; if (this.y < 0) this.y = height; if (this.y > height) this.y = 0; } draw() { noStroke(); fill(this.hue, 30, 90, 0.6); circle(this.x, this.y, this.size); } } function setup() { createCanvas(windowWidth, windowHeight); colorMode(HSB, 360, 100, 100, 1); // create initial chaos for (let i = 0; i < 800; i++) { particles.push(new Particle()); } } function draw() { background(220, 40, 8, 0.07); // very slow fade — feels like deep water // update & draw particles for (let p of particles) { p.update(); p.draw(); } // HOVER DETECTION let distMoved = dist(mouseX, mouseY, lastMouseX, lastMouseY); if (distMoved < 8) { // mouse/finger is almost still hoverTime++; } else { hoverTime = 0; } lastMouseX = mouseX; lastMouseY = mouseY; // If hovering steadily, particles near cursor begin to calm and gather if (hoverTime > 25) { for (let p of particles) { let d = dist(p.x, p.y, mouseX, mouseY); if (d < 180) { // radius of influence // gently pull toward cursor + slow down p.vx = lerp(p.vx, (mouseX - p.x) * 0.008, 0.08); p.vy = lerp(p.vy, (mouseY - p.y) * 0.008, 0.08); // calm chaotic movement over time p.vx *= 0.96; p.vy *= 0.96; // visual feedback — particles brighten when forming p.hue = lerp(p.hue, 140, 0.02); // shift toward soft green (life emerging) } } } // very light repel if moving too fast (chaos returns) if (hoverTime < 5 && distMoved > 30) { for (let p of particles) { let d = dist(p.x, p.y, mouseX, mouseY); if (d < 120) { p.vx += (p.x - mouseX) * 0.015; p.vy += (p.y - mouseY) * 0.015; } } } } function windowResized() { resizeCanvas(windowWidth, windowHeight); }
